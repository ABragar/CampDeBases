CREATE VIEW brut.V_NewsletterContenu 
with SCHEMABINDING
AS

SELECT ce.ConsentementID
	  ,masterId
      ,MarqueID
      ,ce.ContenuID
      ,ce.ConsentementDate
      ,ce.valeur
FROM   brut.ConsentementsEmail AS ce
       INNER JOIN ref.Contenus cn
            ON  cn.TypeContenu = 1
                AND ce.ContenuID = cn.ContenuID
GO
